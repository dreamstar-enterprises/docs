Measures - Product & Customer

No. of Customers who bought a Product

DAX 
No. of Customers who bought a Product =
IF (
    [Same Product Selected],
    BLANK (),
    DISTINCTCOUNT ( Sales_Table[Customer] )
)

No. of Customers who bought both Product 2

DAX 
No. of Customers who bought both Product 2 =
VAR InitialPurchase =
    VALUES ( Sales_Table[Customer] )
VAR ComparisonPurchase =
    CALCULATETABLE (
        VALUES ( Sales_Table[Customer] ),
        ALL ( Products_Table ),
        TREATAS ( VALUES ( 'Filter Products'[Products] ), Sales_Table[Product] )
    )
RETURN
    IF (
        SELECTEDVALUE ( Products_Table[Products] )
            = SELECTEDVALUE ( 'Filter Products'[Products] ),
        BLANK (),
        COUNTROWS ( INTERSECT ( InitialPurchase, ComparisonPurchase ) )
    )

No. of Customers who bought both Products

DAX 
No. of Customers who bought both Products =
VAR MultiPurchaseEvaluation =
    CALCULATETABLE (
        VALUES ( Sales_Table[Customer] ),
        ALL ( Products_Table ),
        USERELATIONSHIP ( Sales_Table[Product], 'Filter Products'[Products] )
    )
RETURN
    IF (
        [Same Product Selected] = TRUE,
        BLANK (),
        CALCULATE ( [No. of Customers who bought a Product], MultiPurchaseEvaluation )
    )

No. of Customers who bought both Products as a %

DAX 
No. of Customers who bought both Products as a % =
DIVIDE (
    [No. of Customers who bought both Products],
    [No. of Customers who bought a Product],
    0
)

Same Product Selected

DAX 
Same Product Selected =
IF (
    SELECTEDVALUE ( 'Filter Products'[Products] )
        <> SELECTEDVALUE ( Sales_Table[Product] ),
    BLANK (),
    TRUE
)
