DAX 
[Actuals: P&L ALL [$]]] = 
VAR ReportingType = "CA=ACTUAL"
RETURN
IF (
    ISFILTERED ( 'Time Period'[Period] ),
    IF (
        HASONEVALUE ( 'Time Period'[Period] ),
        SWITCH (
            VALUES ( 'Time Period'[Period] ),
            "MTD", CALCULATE (
                SUM ( FactPnL[Value] ),
                FactPnL[Reporting Type] = ReportingType,
                DATESMTD ( Calendar_Table[Date] )
            ),
            "QTD", CALCULATE (
                SUM ( FactPnL[Value] ),
                FactPnL[Reporting Type] = ReportingType,
                DATESQTD ( Calendar_Table[Date] )
            ),
            "YTD", CALCULATE (
                SUM ( FactPnL[Value] ),
                FactPnL[Reporting Type] = ReportingType,
                DATESYTD ( Calendar_Table[Date] )
            ),
            "Last Quarter", CALCULATE (
                SUM ( FactPnL[Value] ),
                FactPnL[Reporting Type] = ReportingType,
                DATESINPERIOD (
                    Calendar_Table[Date],
                    MAX ( Calendar_Table[Date] ),
                    VALUES ( 'Time Period'[Offset] ),
                    DAY
                )
            ),
            "Last 12 Months", CALCULATE (
                SUM ( FactPnL[Value] ),
                FactPnL[Reporting Type] = ReportingType,
                DATESINPERIOD (
                    Calendar_Table[Date],
                    MAX ( Calendar_Table[Date] ),
                    VALUES ( 'Time Period'[Offset] ),
                    MONTH
                )
            ),
            CALCULATE (
                SUM ( FactPnL[Value] ),
                FactPnL[Reporting Type] = ReportingType
            )
        )
    ),
    CALCULATE (
        SUM ( FactPnL[Value] ),
        FactPnL[Reporting Type] = ReportingType
    )

VAR ReportingType = "CA=ACTUAL"
VAR Account = "AC=5000"
RETURN
IF (
    OR (
        AND (
            DimHeaderRatios[HeaderRatios Detail] = 0,
            ISFILTERED ( DimAccounts[SubHeader] )
        ),
        AND ( [Sub Header Detail] = 0, ISFILTERED ( DimAccounts[SubHeader 2] ) )
    ),
    BLANK (),
    SWITCH (
        DimHeaderRatios[HeaderRatios CalcType],
        BLANK (), BLANK (),
        3, IFERROR (
            SUMX (
                DimAccounts,
                CALCULATE (
                    SUM ( FactPnL[Value] ),
                    FactPnL[Reporting Type] = ReportingType,
                    DimAccounts[Include in Calcs?] = "Yes"
                )
                    * DimAccounts[Report Sign]
            )
                / CALCULATE (
                    SUMX (
                        DimAccounts,
                        CALCULATE (
                            SUM ( FactPnL[Value] ),
                            FactPnL[Reporting Type] = ReportingType,
                            FactPnL[Account] = Account,
                            DimAccounts[Include in Calcs?] = "Yes"
                        )
                            * DimAccounts[Report Sign]
                    ),
                    ALL ( DimHeaderRatios )
                ),
            BLANK()
        ),
        4, IFERROR (
            CALCULATE (
                SUMX (
                    DimAccounts,
                    CALCULATE (
                        SUM ( FactPnL[Value] ),
                        FactPnL[Reporting Type] = ReportingType,
                        DimAccounts[Include in Calcs?] = "Yes"
                    )
                        * DimAccounts[Sign]
                ),
                ALL ( DimHeaderRatios ),
                DimHeaderRatios[Order] < VALUES ( DimHeaderRatios[Order] )
            )
                / CALCULATE (
                    SUMX (
                        DimAccounts,
                        CALCULATE (
                            SUM ( FactPnL[Value] ),
                            FactPnL[Reporting Type] = ReportingType,
                            FactPnL[Account] = Account,
                            DimAccounts[Include in Calcs?] = "Yes"
